# Podcasts V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a separate Podcasts section where users can subscribe to podcast RSS feeds, discover curated shows, stream episodes, resume progress, star/save episodes, and keep podcasts out of article views.

**Architecture:** Add separate podcast Prisma models and podcast-specific service modules. Reuse the existing app patterns for auth, server actions, reader shell navigation, BullMQ refresh scheduling, collection pickers, and focused Vitest coverage. Podcast audio streams from original enclosure URLs; Arctic RSS stores metadata and user state only.

**Tech Stack:** Next.js App Router, React client components for playback, Prisma/Postgres, BullMQ/Redis worker refresh, Vitest, fast-xml-parser, existing UI primitives.

import { getPrisma } from "./db"

export const FREE_PLAN_SOURCE_LIMIT = 200

type SourceLimitStore = {
  user: {
    findUnique(args: {
      select: {
        _count: {
          select: {
            podcastSubscriptions: true
            subscriptions: true
          }
        }
        plan: true
      }
      where: { id: string }
    }): Promise<{
      _count: {
        podcastSubscriptions: number
        subscriptions: number
      }
      plan: string
    } | null>
  }
}

export class SourceLimitError extends Error {
  constructor(message: string) {
    super(message)
    this.name = "SourceLimitError"
  }
}

export async function assertUserCanAddSource({
  store = getPrisma() as unknown as SourceLimitStore,
  userId,
}: {
  store?: SourceLimitStore
  userId: string
}) {
  const userPlan = await store.user.findUnique({
    where: { id: userId },
    select: {
      plan: true,
      _count: {
        select: {
          subscriptions: true,
          podcastSubscriptions: true,
        },
      },
    },
  })

  if (!userPlan) {
    throw new SourceLimitError("User account was not found.")
  }

  const currentSourceCount =
    userPlan._count.subscriptions + userPlan._count.podcastSubscriptions

  if (userPlan.plan === "FREE" && currentSourceCount >= FREE_PLAN_SOURCE_LIMIT) {
    throw new SourceLimitError(
      `Free accounts can subscribe to up to ${FREE_PLAN_SOURCE_LIMIT} sources.`
    )
  }

  return {
    currentSourceCount,
    limit: FREE_PLAN_SOURCE_LIMIT,
  }
}
```

import { XMLParser } from "fast-xml-parser"

export type ParsedPodcastEpisode = {
  audioLengthBytes?: bigint
  audioType?: string
  audioUrl: string
  contentHtml?: string
  contentText?: string
  description?: string
  durationSeconds?: number
  externalId: string
  imageUrl?: string
  publishedAt?: Date
  title: string
  url?: string
}

export type ParsedPodcastFeed = {
  artworkUrl?: string
  author?: string
  description?: string
  episodes: ParsedPodcastEpisode[]
  feedUrl: string
  language?: string
  siteUrl?: string
  title: string
}

export class PodcastParseError extends Error {
  constructor(message: string) {
    super(message)
    this.name = "PodcastParseError"
  }
}

const parser = new XMLParser({
  attributeNamePrefix: "",
  ignoreAttributes: false,
  trimValues: true,
})

export function parsePodcastFeed(xml: string, feedUrl: string): ParsedPodcastFeed {
  const document = parser.parse(xml)
  const channel = document?.rss?.channel ?? document?.feed

  if (!channel) {
    throw new PodcastParseError("Podcast RSS channel was not found.")
  }

  const items = asArray(channel.item ?? channel.entry)
  const episodes = items.map(parseEpisode).filter(Boolean) as ParsedPodcastEpisode[]

  if (episodes.length === 0) {
    throw new PodcastParseError("No audio episodes were found in that podcast feed.")
  }

  return {
    artworkUrl: attr(channel["itunes:image"], "href") ?? text(channel.image?.url),
    author: text(channel["itunes:author"]) ?? text(channel.author?.name),
    description: text(channel.description) ?? text(channel.subtitle),
    episodes,
    feedUrl,
    language: text(channel.language),
    siteUrl: text(channel.link),
    title: text(channel.title) ?? "Untitled Podcast",
  }
}

function parseEpisode(item: Record<string, unknown>): ParsedPodcastEpisode | null {
  const enclosure = item.enclosure as Record<string, unknown> | undefined
  const audioUrl = attr(enclosure, "url")

  if (!audioUrl) {
    return null
  }

  const title = text(item.title) ?? "Untitled Episode"
  const url = text(item.link)

  return {
    audioLengthBytes: parseBigInt(attr(enclosure, "length")),
    audioType: attr(enclosure, "type"),
    audioUrl,
    contentHtml: text(item["content:encoded"]),
    contentText: text(item.description),
    description: text(item.description),
    durationSeconds: parseDuration(text(item["itunes:duration"])),
    externalId: text(item.guid) ?? url ?? audioUrl,
    imageUrl: attr(item["itunes:image"] as Record<string, unknown> | undefined, "href"),
    publishedAt: parseDate(text(item.pubDate) ?? text(item.published)),
    title,
    url,
  }
}

function asArray<T>(value: T | T[] | undefined): T[] {
  if (!value) return []
  return Array.isArray(value) ? value : [value]
}

function attr(value: Record<string, unknown> | undefined, key: string) {
  const raw = value?.[key]
  return typeof raw === "string" && raw.trim() ? raw.trim() : undefined
}

function text(value: unknown): string | undefined {
  if (typeof value === "string") return value.trim() || undefined
  if (typeof value === "number") return String(value)
  if (value && typeof value === "object" && "#text" in value) {
    const raw = (value as { "#text"?: unknown })["#text"]
    return typeof raw === "string" ? raw.trim() || undefined : undefined
  }
  return undefined
}

function parseBigInt(value?: string) {
  if (!value || !/^\d+$/.test(value)) return undefined
  return BigInt(value)
}

function parseDate(value?: string) {
  if (!value) return undefined
  const date = new Date(value)
  return Number.isNaN(date.getTime()) ? undefined : date
}

function parseDuration(value?: string) {
  if (!value) return undefined
  if (/^\d+$/.test(value)) return Number(value)
  const parts = value.split(":").map((part) => Number(part))
  if (parts.some((part) => !Number.isFinite(part))) return undefined
  return parts.reduce((total, part) => total * 60 + part, 0)
}
```

import { getPrisma } from "./db"
import { parsePodcastFeed, type ParsedPodcastEpisode } from "./podcast-parser"
import {
  normalizeHttpUrl,
  safeFetchText,
  type SafeFetchTextResult,
} from "./url-safety"

type RefreshablePodcast = {
  feedUrl: string
  id: string
}

type PodcastRefreshStore = {
  podcast: {
    findUnique(args: {
      select: { feedUrl: true; id: true }
      where: { id: string }
    }): Promise<RefreshablePodcast | null>
    update(args: {
      data: Record<string, unknown>
      where: { id: string }
    }): Promise<unknown>
  }
  podcastEpisode: {
    upsert(args: {
      create: Record<string, unknown>
      update: Record<string, unknown>
      where: {
        podcastId_externalId: {
          externalId: string
          podcastId: string
        }
      }
    }): Promise<unknown>
  }
}

export type PodcastRefreshResult = {
  episodeCount: number
  podcastId: string
}

export class PodcastRefreshError extends Error {
  constructor(message: string) {
    super(message)
    this.name = "PodcastRefreshError"
  }
}

export async function refreshPodcast(podcastId: string) {
  return refreshPodcastWithClient({
    podcastId,
    store: getPodcastRefreshStore(),
  })
}

export async function refreshPodcastWithClient({
  fetchText = safeFetchText,
  now = () => new Date(),
  podcastId,
  store = getPodcastRefreshStore(),
}: {
  fetchText?: (url: URL) => Promise<SafeFetchTextResult>
  now?: () => Date
  podcastId: string
  store?: PodcastRefreshStore
}): Promise<PodcastRefreshResult> {
  const podcast = await store.podcast.findUnique({
    select: { feedUrl: true, id: true },
    where: { id: podcastId },
  })

  if (!podcast) {
    throw new PodcastRefreshError("Podcast not found.")
  }

  const fetchedAt = now()

  try {
    const response = await fetchText(normalizeHttpUrl(podcast.feedUrl))
    const parsedPodcast = parsePodcastFeed(response.text, response.url.href)

    for (const episode of parsedPodcast.episodes) {
      await store.podcastEpisode.upsert(episodeUpsertArgs(podcast.id, episode))
    }

    await store.podcast.update({
      data: {
        artworkUrl: parsedPodcast.artworkUrl,
        author: parsedPodcast.author,
        description: parsedPodcast.description,
        language: parsedPodcast.language,
        lastError: null,
        lastFailedAt: null,
        lastFetchedAt: fetchedAt,
        lastSuccessfulFetchAt: fetchedAt,
        siteUrl: parsedPodcast.siteUrl,
        title: parsedPodcast.title,
      },
      where: { id: podcast.id },
    })

    return {
      episodeCount: parsedPodcast.episodes.length,
      podcastId: podcast.id,
    }
  } catch (error) {
    await store.podcast.update({
      data: {
        lastError: errorMessage(error),
        lastFailedAt: fetchedAt,
        lastFetchedAt: fetchedAt,
      },
      where: { id: podcast.id },
    })

    throw error
  }
}

function episodeUpsertArgs(podcastId: string, episode: ParsedPodcastEpisode) {
  return {
    create: withoutUndefined({ ...episode, podcastId }),
    update: withoutUndefined({
      audioLengthBytes: episode.audioLengthBytes,
      audioType: episode.audioType,
      audioUrl: episode.audioUrl,
      contentHtml: episode.contentHtml,
      contentText: episode.contentText,
      description: episode.description,
      durationSeconds: episode.durationSeconds,
      imageUrl: episode.imageUrl,
      publishedAt: episode.publishedAt,
      title: episode.title,
      url: episode.url,
    }),
    where: {
      podcastId_externalId: {
        externalId: episode.externalId,
        podcastId,
      },
    },
  }
}

function withoutUndefined(values: Record<string, unknown>) {
  return Object.fromEntries(
    Object.entries(values).filter(([, value]) => value !== undefined)
  )
}

function errorMessage(error: unknown) {
  if (error instanceof Error && error.message) {
    return error.message.slice(0, 500)
  }
  return "Podcast refresh failed."

export async function subscribeToPodcast({
  url,
  userId,
}: {
  url: string
  userId: string
}) {
  const prisma = getPrisma()
  const normalizedUrl = normalizeHttpUrl(url)

  try {
    await assertUserCanAddSource({ userId })
  } catch (error) {
    if (error instanceof SourceLimitError) {
      throw new PodcastSubscriptionError(error.message)
    }
    throw error
  }

  const response = await safeFetchText(normalizedUrl)
  const parsedPodcast = parsePodcastFeed(response.text, response.url.href)

  const existingSubscription = await prisma.podcastSubscription.findFirst({
    where: {
      userId,
      podcast: {
        feedUrl: parsedPodcast.feedUrl,
      },
    },
    select: {
      podcast: {
        select: {
          title: true,
        },
      },
    },
  })

  if (existingSubscription) {
    throw new PodcastSubscriptionError(
      `You are already subscribed to ${existingSubscription.podcast.title}.`
    )
  }

  const now = new Date()
  const podcast = await prisma.podcast.upsert({
    where: { feedUrl: parsedPodcast.feedUrl },
    create: {
      artworkUrl: parsedPodcast.artworkUrl,
      author: parsedPodcast.author,
      description: parsedPodcast.description,
      feedUrl: parsedPodcast.feedUrl,
      language: parsedPodcast.language,
      lastError: null,
      lastFetchedAt: now,
      lastFailedAt: null,
      lastSuccessfulFetchAt: now,
      siteUrl: parsedPodcast.siteUrl,
      title: parsedPodcast.title,
    },
    update: {
      artworkUrl: parsedPodcast.artworkUrl,
      author: parsedPodcast.author,
      description: parsedPodcast.description,
      language: parsedPodcast.language,
      lastError: null,
      lastFetchedAt: now,
      lastFailedAt: null,
      lastSuccessfulFetchAt: now,
      siteUrl: parsedPodcast.siteUrl,
      title: parsedPodcast.title,
    },
  })

  let subscription
  try {
    subscription = await prisma.podcastSubscription.create({
      data: {
        podcastId: podcast.id,
        userId,
      },
      include: {
        podcast: true,
      },
    })
  } catch (error) {
    if (
      error instanceof Prisma.PrismaClientKnownRequestError &&
      error.code === "P2002"
    ) {
      throw new PodcastSubscriptionError(
        `You are already subscribed to ${podcast.title}.`
      )
    }
    throw error
  }

  let initialEpisodeCount: number | undefined
  try {
    const result = await refreshPodcastWithClient({
      podcastId: podcast.id,
      fetchText: async () => response,
    })
    initialEpisodeCount = result.episodeCount
  } catch {
    initialEpisodeCount = undefined
  }

  return {
    ...subscription,
    initialEpisodeCount,
  }
}

export async function getPodcastHomeForUser(userId: string) {
  const prisma = getPrisma()
  const [subscriptions, episodes, unplayedCounts] = await Promise.all([
    prisma.podcastSubscription.findMany({
      where: { userId },
      include: {
        podcast: {
          include: {
            episodes: {
              orderBy: { publishedAt: "desc" },
              take: 1,
            },
          },
        },
      },
      orderBy: [{ sortOrder: "asc" }, { subscribedAt: "desc" }],
    }),
    prisma.podcastEpisode.findMany({
      where: {
        podcast: {
          subscriptions: {
            some: {
              isPaused: false,
              userId,
            },
          },
        },
      },
      include: {
        podcast: true,
        states: {
          where: { userId },
          take: 1,
        },
      },
      orderBy: [{ publishedAt: "desc" }, { createdAt: "desc" }],
      take: 100,
    }),
    prisma.podcastEpisode.groupBy({
      by: ["podcastId"],
      _count: { _all: true },
      where: {
        podcast: {
          subscriptions: {
            some: {
              isPaused: false,
              userId,
            },
          },
        },
        OR: [
          { states: { none: { userId } } },
          { states: { some: { userId, isPlayed: false } } },
        ],
      },
    }),
  ])

  const unplayedCountByPodcastId = new Map(
    unplayedCounts.map((row) => [row.podcastId, row._count._all])
  )

  return {
    episodes: episodes.map(mapEpisode),
    subscriptions: subscriptions.map((subscription) => ({
      artworkUrl: subscription.podcast.artworkUrl,
      id: subscription.podcast.id,
      lastError: subscription.podcast.lastError,
      latestEpisodeTitle: subscription.podcast.episodes[0]?.title ?? null,
      subscriptionId: subscription.id,
      title: subscription.customTitle || subscription.podcast.title,
      unplayedCount: unplayedCountByPodcastId.get(subscription.podcastId) ?? 0,
    })),
  }
}

function mapEpisode(
  episode: Prisma.PodcastEpisodeGetPayload<{
    include: {
      podcast: true
      states: true
    }
  }>
) {
  const state = episode.states[0]

  return {
    audioMimeType: episode.audioMimeType,
    audioUrl: episode.audioUrl,
    description: episode.description,
    durationSeconds: episode.durationSeconds,
    episodeId: episode.id,
    imageUrl: episode.imageUrl || episode.podcast.artworkUrl,
    isPlayed: state?.isPlayed ?? false,
    isStarred: state?.isStarred ?? false,
    playbackPositionSeconds: state?.playbackPositionSeconds ?? 0,
    podcastId: episode.podcastId,
    podcastTitle: episode.podcast.title,
    publishedAt: episode.publishedAt,
    title: episode.title,
    url: episode.url,

}

export async function getPodcastShowForUser({
  podcastId,
  userId,
}: {
  podcastId: string
  userId: string
}) {
  const podcast = await getPrisma().podcast.findFirst({
    where: {
      id: podcastId,
      subscriptions: {
        some: {
          userId,
        },
      },
    },
    include: {
      episodes: {
        include: {
          states: {
            where: { userId },
            take: 1,
          },
        },
        orderBy: [{ publishedAt: "desc" }, { createdAt: "desc" }],
      },
    },
  })

  if (!podcast) return null

  return {
    podcast: {
      artworkUrl: podcast.artworkUrl,
      description: podcast.description,
      id: podcast.id,
      lastError: podcast.lastError,
      title: podcast.title,
      url: podcast.siteUrl,
    },
    episodes: podcast.episodes.map((episode) =>
      mapEpisode({ ...episode, podcast, states: episode.states })
    ),
  }
}
